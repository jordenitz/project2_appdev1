import java.util.List;
import java.util.Map;

public class Member {
    private Long userId;      // Telegram user id
    private String name;      // first + last or username
}

public class Survey {
    private Long id;
    private Long creatorId;   // מי יצר את הסקר (אולי לא חובה אם רק הבוט מנהל)
    private List<Question> questions;
    private boolean active;
    private long startTimeMillis;
    private long endTimeMillis;
    private Map<Long, UserSurveyAnswer> answersByUser; // userId -> תשובות לכל השאלות
}

public class Question {
    private int index; // 0,1,2
    private String text;
    private List<AnswerOption> options;
}

public class AnswerOption {
    private int index; // 0..3
    private String text;
}

public class UserSurveyAnswer {
    private Long userId;
    // לכל שאלה מה האינדקס של התשובה שנבחרה
    private Map<Integer, Integer> selectedOptionByQuestionIndex;
}
